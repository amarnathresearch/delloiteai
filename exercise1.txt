#include<stdio.h>
int main(){	
	int i, n=10;
	for(i=0; i<n;i++){
	
	    if(i == 5){
	         break;
	    }
	}
	return 0;
}
n= 10, 1 unit, 1 iteration
i = 0, 1 unit, 1 iteration


i < n; 1 unit,  5 times

i = 0,1, 2, 3, 4,5, 6 times

i++, i=1,2, 3,4,5, 5 times,

i == 5, 6 times it will get executed

i++, i=i+1; 2 units, 4 times

i == 5, 1 unit, 6 times

T(n) = O(n)
